from wtforms.validators import InputRequired,Optional,Email

class NewEmployeeForm(FlaskForm):

    name=StringField('Name',
                    validators=[InputRequired])
    
    email=StringField('Email Address'.
                      validators=[Optional(),Email() ]  )
